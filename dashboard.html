INVESTMENT EXECUTION async function startPlan(name, multiplier, inputId) { const amtInput = document.getElementById(inputId); const amount = parseFloat(amtInput.value) || 150; if (amount < 150) { alert("Minimum investment is $150."); return; } if (amount > userBalance) { alert("Insufficient funds in available USD balance."); return; } const expectedReturn = amount * multiplier; const confirmAction = confirm(`Confirm $${amount.toLocaleString()} investment in ${name} Plan?\nExpected Return: $${expectedReturn.toLocaleString()} in 7 days.`); if (confirmAction) { try { // Silently deduct from balance const response = await fetch(`${api}/api/user/update-balance`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email: email, amount: -amount }) }); if (response.ok) { // Log to History const investmentData = { plan: name, amount: amount, return: expectedReturn, date: new Date().toLocaleDateString(), timestamp: Date.now(), status: 'Awaiting Investment Return' }; saveToHistory(investmentData); alert(`Investment Successful! Your return of $${expectedReturn.toLocaleString()} will be credited in 7 days.`); sync(); // Update UI balance } } catch (err) { console.error("Investment Error:", err); } } } // 2. `; sells.innerHTML += `

`; } } // 3.
